def caesar_cipher(text, shift, mode):
    """
    Encrypts or decrypts text using the Caesar Cipher algorithm.
    
    Args:
        text (str): The text to process
        shift (int): The number of positions to shift letters
        mode (str): 'encrypt' or 'decrypt'
    
    Returns:
        str: The processed text
    """
    if mode == 'decrypt':
        shift = -shift
    
    result = ""
    
    for char in text:
        if char.isalpha():
            
            ascii_offset = 65 if char.isupper() else 97
            
            
            shifted_position = (ord(char) - ascii_offset + shift) % 26
            new_char = chr(shifted_position + ascii_offset)
            result += new_char
        else:
            
            result += char
    
    return result

def main():
    print("% Caesar Cipher Program %")
    print("This program encrypts and decrypts text using the Caesar Cipher.")
    
    while True:
        print("\nOptions:")
        print("1. Encrypt a message")
        print("2. Decrypt a message")
        print("3. Exit")
        
        choice = input("\nEnter your choice (1-3): ").strip()
        
        if choice == '1':
            message = input("Enter the message to encrypt: ")
            try:
                shift = int(input("Enter the shift value (1-25): "))
                if shift < 1 or shift > 25:
                    print("Shift value must be between 1 and 25. Using 3 as default.")
                    shift = 3
            except ValueError:
                print("Invalid shift value. Using 3 as default.")
                shift = 3
            
            encrypted = caesar_cipher(message, shift, 'encrypt')
            print(f"\nOriginal message: {message}")
            print(f"Encrypted message: {encrypted}")
            print(f"Shift used: {shift}")
        
        elif choice == '2':
            message = input("Enter the message to decrypt: ")
            try:
                shift = int(input("Enter the shift value used for encryption: "))
                if shift < 1 or shift > 25:
                    print("Shift value must be between 1 and 25. Using 3 as default.")
                    shift = 3
            except ValueError:
                print("Invalid shift value. Using 3 as default.")
                shift = 3
            
            decrypted = caesar_cipher(message, shift, 'decrypt')
            print(f"\nEncrypted message: {message}")
            print(f"Decrypted message: {decrypted}")
            print(f"Shift used: {shift}")
        
        elif choice == '3':
            print("Thank you for using the Caesar Cipher program!")
            break
        
        else:
            print("Invalid choice. Please enter 1, 2, or 3.")

if __name__ == "__main__":
    main()
